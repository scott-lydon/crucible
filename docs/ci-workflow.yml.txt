name: CI

on:
  push:
    branches: [main, "julian/**"]
  pull_request:

jobs:
  quality:
    runs-on: ubuntu-latest
    services:
      postgres:
        image: postgres:16-alpine
        env:
          POSTGRES_USER: crucible
          POSTGRES_PASSWORD: crucible
          POSTGRES_DB: crucible
        ports: ["5432:5432"]
        options: >-
          --health-cmd "pg_isready -U crucible"
          --health-interval 5s
          --health-timeout 5s
          --health-retries 15
    env:
      PGHOST: localhost
      PGPORT: "5432"
      PGUSER: crucible
      PGPASSWORD: crucible
      DATABASE_URL: postgresql+asyncpg://crucible:crucible@localhost:5432/crucible
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with:
          python-version: "3.12"
      - name: Install
        run: pip install -e ".[dev]"
      - name: Ruff
        run: ruff check .
      - name: Mypy (strict)
        run: mypy shared orchestrator modules scripts
      - name: Architecture check
        run: python scripts/check_module_imports.py
      - name: Migrations apply
        run: alembic upgrade head
      - name: Tests
        run: pytest -q --cov=shared --cov=orchestrator --cov=modules --cov-report=term-missing
